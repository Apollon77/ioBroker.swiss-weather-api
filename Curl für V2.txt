-- Dies muss immer gemacht werden, da nicht für jede Koordinate auch eine vorhersage besteht
-- Deswegen muss zuerst die nächste Geolocationid mit vorhandener Vorhersage ermittelt werden.

Request URL: https://api.srgssr.ch/srf-meteo/v2/geolocations?latitude=47.0461&longitude=7.3756

curl -X 'GET' \
  'https://api.srgssr.ch/srf-meteo/v2/geolocations?latitude=47.0461&longitude=7.3756' \
  -H 'accept: application/json' \
  -H 'authorization: Bearer azr8BxAD2pPdnpvOMRCo7VKqehGW'
